{{- $u := .Destination -}} {{- $imgUrl := "" -}} {{- if or (strings.HasPrefix $u "/assets/") (strings.HasPrefix $u "assets/") -}} {{- $cleanPath := strings.TrimPrefix "assets/" (strings.TrimPrefix "/assets/" $u) -}} {{- /* 1. Try Hugo Pipes (assets/ directory) */ -}} {{- $img := resources.Get $cleanPath -}} {{- if $img -}} {{- $imgUrl = $img.RelPermalink -}} {{- else -}} {{- /* 2. Fall back to static/ directory (path without "assets/" prefix) */ -}} {{- $staticPath := printf "static/%s" $cleanPath -}} {{- if fileExists $staticPath -}} {{- $imgUrl = printf "/%s" $cleanPath | relURL -}} {{- end -}} {{- end -}} {{- end -}} {{- /* 3. Final fallback: use the URL as-is (relURL handles subpath) */ -}} {{- if not $imgUrl -}} {{- $imgUrl = $u | relURL -}} {{- end -}}
